function panelToggle(panelIdString, closePanel1String, closePanel2String) {
	
	if (panelIdString) {	
		var panelId = document.getElementById(panelIdString);
		var panelDisplay = window.getComputedStyle(panelId, null).getPropertyValue('display');

		if (panelDisplay === 'none' || panelDisplay === '') {
			panelId.style.display = 'block';
		} else {
			panelId.style.display = 'none';
		}
	}
  
	if (closePanel1String) {
		var closePanel1Id = document.getElementById(closePanel1String);

		closePanel1Id.style.display = 'none';
	}

	if (closePanel2String) {
		var closePanel2Id = document.getElementById(closePanel2String);

		closePanel2Id.style.display = 'none';
	}
}

function closeSlidePanel (closePanelString) {
		var closePanelId = document.getElementById(closePanelString);
		var panelLeft = window.getComputedStyle(closePanelId, null).getPropertyValue('left');
		var panelWidth = parseInt(window.getComputedStyle(closePanelId, null).getPropertyValue('width'));
		var panelPaddingLeft = parseInt(window.getComputedStyle(closePanelId, null).getPropertyValue('padding-left'));
		var panelOffset = panelWidth + panelPaddingLeft;

		closePanelId.style.left = "-"+String(panelOffset)+"px";
}

/* Toggle the opening or closing of a slider panel, and closes additional panels*/
function panelSlide(panelIdString, closePanel1String, closePanel2String) {
	
	if (panelIdString) {
		var panelId = document.getElementById(panelIdString);
		var panelLeft = window.getComputedStyle(panelId, null).getPropertyValue('left');

		if (panelLeft === "0px" || panelLeft === "") {
			closeSlidePanel (panelIdString);
		} else {
			panelId.style.left = "0";
		}
	}
  
	if (closePanel1String) {
		closeSlidePanel (closePanel1String);
	}
  
	if (closePanel2String) {
		closeSlidePanel (closePanel2String);
	}
}
